Create a Google Sheet

// MASRAF - Google Apps Script Backend

const SHEET_NAME          = 'Spending';
const INCOME_SHEET_NAME   = 'Income';
const SCHEDULED_SHEET_NAME = 'Scheduled';
const SAVINGS_SHEET_NAME  = 'Savings';

// ── SHEET HELPERS ─────────────────────────────────────────────────────────────

function normalizeSheetDate(date) {
  if (date instanceof Date) {
    const d = String(date.getDate()).padStart(2,'0');
    const m = String(date.getMonth()+1).padStart(2,'0');
    return d + '/' + m + '/' + date.getFullYear();
  }
  return String(date);
}

function styleHeader(sheet, cols, color) {
  const header = sheet.getRange(1, 1, 1, cols);
  header.setBackground('#0f0f0f');
  header.setFontColor(color);
  header.setFontWeight('bold');
}

function getOrCreateSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
    sheet.appendRow(['ID','Date','Category','Amount (LL)','Amount ($)','Timestamp']);
    sheet.setFrozenRows(1);
    styleHeader(sheet, 6, '#c8f060');
    [80,80,180,120,100,160].forEach((w,i) => sheet.setColumnWidth(i+1, w));
  }
  return sheet;
}

function getOrCreateIncomeSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(INCOME_SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(INCOME_SHEET_NAME);
    sheet.appendRow(['ID','Date','Description','Amount (LL)','Amount ($)','Timestamp']);
    sheet.setFrozenRows(1);
    styleHeader(sheet, 6, '#60d4f0');
    [80,80,180,120,100,160].forEach((w,i) => sheet.setColumnWidth(i+1, w));
  }
  return sheet;
}

function getOrCreateScheduledSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SCHEDULED_SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SCHEDULED_SHEET_NAME);
    sheet.appendRow(['ID','Month','Description','Amount (LL)','Amount ($)','Status','Spent Date','Spent Entry ID']);
    sheet.setFrozenRows(1);
    styleHeader(sheet, 8, '#f0a060');
    [80,80,180,120,100,80,100,120].forEach((w,i) => sheet.setColumnWidth(i+1, w));
  }
  // Force Month column (B) to plain text so Google Sheets never auto-converts MM/YYYY to a Date
  sheet.getRange('B:B').setNumberFormat('@');
  return sheet;
}

// ── ROUTING ───────────────────────────────────────────────────────────────────

function doGet(e) {
  const action = e.parameter.action;
  if (action === 'getAll')       return getAll();
  if (action === 'getIncome')    return getIncome();
  if (action === 'getScheduled') return getScheduled();
  if (action === 'getSavings')   return getSavings();
  if (action === 'getRate')      return getRate();
  return jsonResponse({ error: 'Unknown action' });
}

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    if (body.action === 'add')             return addEntry(body);
    if (body.action === 'addBatch')        return addBatch(body);
    if (body.action === 'delete')          return deleteEntry(body);
    if (body.action === 'addIncome')       return addIncome(body);
    if (body.action === 'deleteIncome')    return deleteIncome(body);
    if (body.action === 'addScheduled')     return addScheduled(body);
    if (body.action === 'spendScheduled')   return spendScheduled(body);
    if (body.action === 'deleteScheduled')  return deleteScheduled(body);
    if (body.action === 'restoreScheduled') return restoreScheduled(body);
    if (body.action === 'addSavings')        return addSavings(body);
    if (body.action === 'withdrawSaving')   return withdrawSaving(body);
    if (body.action === 'deleteSaving')     return deleteSaving(body);
    if (body.action === 'deleteWithdrawal') return deleteWithdrawal(body);
    if (body.action === 'setRate')         return setRate(body);
    return jsonResponse({ error: 'Unknown action' });
  } catch(err) {
    return jsonResponse({ error: err.message });
  }
}

// ── SPENDING ──────────────────────────────────────────────────────────────────

function getAll() {
  const sheet = getOrCreateSheet();
  const rows  = sheet.getDataRange().getValues();
  const entries = [];
  for (let i = 1; i < rows.length; i++) {
    if (rows[i][0]) entries.push({
      id: rows[i][0], date: normalizeSheetDate(rows[i][1]),
      cat: rows[i][2], amt: rows[i][3], usd: rows[i][4], row: i + 1
    });
  }
  return jsonResponse({ entries });
}

function addEntry(body) {
  const sheet = getOrCreateSheet();
  const rate  = getRate_().rate;
  const id    = Date.now().toString();
  sheet.appendRow([id, body.date, body.cat, body.amt, (body.amt/rate).toFixed(2), new Date().toISOString()]);
  return jsonResponse({ ok: true, id });
}

function addBatch(body) {
  const sheet = getOrCreateSheet();
  const rate  = getRate_().rate;
  const ids   = [];
  (body.entries || []).forEach(function(entry, i) {
    const id = String(Date.now() + i);
    sheet.appendRow([id, entry.date, entry.cat, entry.amt, (entry.amt/rate).toFixed(2), new Date().toISOString()]);
    ids.push(id);
  });
  return jsonResponse({ ok: true, ids });
}

function deleteEntry(body) {
  const sheet = getOrCreateSheet();
  const rows  = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][0]) === String(body.id)) {
      sheet.deleteRow(i + 1);
      return jsonResponse({ ok: true });
    }
  }
  return jsonResponse({ error: 'Entry not found' });
}

// ── SCHEDULED ─────────────────────────────────────────────────────────────────

function normalizeMonthKey(v) {
  // Google Sheets may auto-convert "MM/YYYY" to a Date — recover it safely
  var d = null;
  if (v instanceof Date) {
    d = v;
  } else {
    var s = String(v).trim();
    // Already MM/YYYY or M/YYYY
    if (/^\d{1,2}\/\d{4}$/.test(s)) {
      var p = s.split('/');
      return (p[0].length === 1 ? '0' + p[0] : p[0]) + '/' + p[1];
    }
    // Might be a Date.toString() — try parsing
    if (s.length > 7) {
      var parsed = new Date(s);
      if (!isNaN(parsed.getTime())) d = parsed;
    }
  }
  if (d) {
    var m = d.getMonth() + 1;
    return (m < 10 ? '0' + m : '' + m) + '/' + d.getFullYear();
  }
  return String(v);
}

function getScheduled() {
  const sheet = getOrCreateScheduledSheet();
  const rows  = sheet.getDataRange().getValues();
  const entries = [];
  for (let i = 1; i < rows.length; i++) {
    if (rows[i][0]) entries.push({
      id:           String(rows[i][0]),
      month:        normalizeMonthKey(rows[i][1]),
      desc:         rows[i][2],
      amt:          rows[i][3],
      status:       rows[i][5] || 'pending',
      spentDate:    rows[i][6] ? normalizeSheetDate(rows[i][6]) : '',
      spentEntryId: rows[i][7] ? String(rows[i][7]) : '',
      row:          i + 1
    });
  }
  return jsonResponse({ entries });
}

function addScheduled(body) {
  const sheet = getOrCreateScheduledSheet();
  const rate  = getRate_().rate;
  const ids   = [];
  (body.entries || []).forEach(function(entry, i) {
    const id = String(Date.now() + i);
    sheet.appendRow([id, entry.month, entry.desc, entry.amt, (entry.amt/rate).toFixed(2), 'pending', '', '']);
    ids.push(id);
  });
  return jsonResponse({ ok: true, ids });
}

function spendScheduled(body) {
  // 1. Add to Spending sheet
  const spendSheet = getOrCreateSheet();
  const rate       = getRate_().rate;
  const spendId    = Date.now().toString();
  spendSheet.appendRow([spendId, body.date, body.desc, body.amt, (body.amt/rate).toFixed(2), new Date().toISOString()]);

  // 2. Update Scheduled row status
  const schedSheet = getOrCreateScheduledSheet();
  const rows       = schedSheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][0]) === String(body.id)) {
      schedSheet.getRange(i + 1, 6).setValue('spent');
      schedSheet.getRange(i + 1, 7).setValue(body.date);
      schedSheet.getRange(i + 1, 8).setValue(spendId);
      break;
    }
  }
  return jsonResponse({ ok: true, spendId });
}

function restoreScheduled(body) {
  // Reset a scheduled entry back to pending (when its spending entry is deleted)
  const sheet = getOrCreateScheduledSheet();
  const rows  = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][0]) === String(body.id)) {
      sheet.getRange(i + 1, 6).setValue('pending');
      sheet.getRange(i + 1, 7).setValue('');
      sheet.getRange(i + 1, 8).setValue('');
      return jsonResponse({ ok: true });
    }
  }
  return jsonResponse({ error: 'Scheduled entry not found' });
}

function deleteScheduled(body) {
  const sheet = getOrCreateScheduledSheet();
  const rows  = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][0]) === String(body.id)) {
      sheet.deleteRow(i + 1);
      return jsonResponse({ ok: true });
    }
  }
  return jsonResponse({ error: 'Scheduled entry not found' });
}

// ── SAVINGS ───────────────────────────────────────────────────────────────────

function getOrCreateSavingsSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SAVINGS_SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SAVINGS_SHEET_NAME);
    sheet.appendRow(['ID','Month','Description','Amount (LL)','Amount ($)','Status','Date','Source Saving ID']);
    sheet.setFrozenRows(1);
    styleHeader(sheet, 8, '#a060f0');
    [80,80,180,120,100,100,120,120].forEach((w,i) => sheet.setColumnWidth(i+1, w));
  }
  // Force Month and Date columns to plain text — prevents Google Sheets auto-date conversion
  sheet.getRange('B:B').setNumberFormat('@');
  sheet.getRange('G:G').setNumberFormat('@');
  sheet.getRange('H:H').setNumberFormat('@');
  return sheet;
}

function getSavings() {
  const sheet = getOrCreateSavingsSheet();
  const rows  = sheet.getDataRange().getValues();
  const entries = [];
  for (var i = 1; i < rows.length; i++) {
    if (!rows[i][0]) continue;
    var status = String(rows[i][5] || 'saving');
    // Normalise: old 'withdrawn' rows from previous schema — treat as 'saving'
    if (status !== 'saving' && status !== 'withdrawal') status = 'saving';
    entries.push({
      id:            String(rows[i][0]),
      month:         normalizeMonthKey(rows[i][1]),
      desc:          rows[i][2],
      amt:           rows[i][3],
      status:        status,
      date:          rows[i][6] ? normalizeSheetDate(rows[i][6]) : '',
      sourceSavingId: rows[i][7] ? String(rows[i][7]) : ''
    });
  }
  return jsonResponse({ entries });
}

function addSavings(body) {
  const sheet = getOrCreateSavingsSheet();
  const rate  = getRate_().rate;
  const ids   = [];
  (body.entries || []).forEach(function(entry, i) {
    const id = String(Date.now() + i);
    sheet.appendRow([id, entry.month, entry.desc, entry.amt, (entry.amt/rate).toFixed(2), 'saving', '', '']);
    ids.push(id);
  });
  return jsonResponse({ ok: true, ids });
}

function withdrawSaving(body) {
  // body: { id (saving row ID), date (DD/MM/YYYY), withdrawnAmt (LL) }
  var sheet = getOrCreateSavingsSheet();
  var rows  = sheet.getDataRange().getValues();
  var rate  = getRate_().rate;
  for (var i = 1; i < rows.length; i++) {
    if (String(rows[i][0]) !== String(body.id)) continue;
    var currentAmt    = Number(rows[i][3]);
    var savedMonth    = String(rows[i][1]);
    var desc          = rows[i][2];
    var withdrawnAmt  = Number(body.withdrawnAmt);
    var isFullWithdraw = withdrawnAmt >= currentAmt;

    // 1. Append withdrawal row BEFORE deleting (avoids index shifts)
    var wId = String(Date.now());
    sheet.appendRow([
      wId, savedMonth, desc,
      withdrawnAmt, (withdrawnAmt / rate).toFixed(2),
      'withdrawal', body.date,
      isFullWithdraw ? '' : body.id   // sourceSavingId empty when fully deleted
    ]);

    if (isFullWithdraw) {
      // 2a. Delete saving row
      sheet.deleteRow(i + 1);
    } else {
      // 2b. Reduce saving amount
      var remaining = currentAmt - withdrawnAmt;
      sheet.getRange(i + 1, 4).setValue(remaining);
      sheet.getRange(i + 1, 5).setValue((remaining / rate).toFixed(2));
    }
    return jsonResponse({ ok: true, withdrawalId: wId, isFullWithdraw: isFullWithdraw });
  }
  return jsonResponse({ error: 'Savings entry not found' });
}

function deleteSaving(body) {
  var sheet = getOrCreateSavingsSheet();
  var rows  = sheet.getDataRange().getValues();
  for (var i = 1; i < rows.length; i++) {
    if (String(rows[i][0]) === String(body.id)) {
      sheet.deleteRow(i + 1);
      return jsonResponse({ ok: true });
    }
  }
  return jsonResponse({ error: 'Savings entry not found' });
}

function deleteWithdrawal(body) {
  // Delete withdrawal row and restore the saving amount (or recreate the saving row)
  var sheet = getOrCreateSavingsSheet();
  var rows  = sheet.getDataRange().getValues();
  var rate  = getRate_().rate;
  for (var i = 1; i < rows.length; i++) {
    if (String(rows[i][0]) !== String(body.id)) continue;
    var withdrawnAmt   = Number(rows[i][3]);
    var savedMonth     = String(rows[i][1]);
    var desc           = rows[i][2];
    var sourceSavingId = rows[i][7] ? String(rows[i][7]) : '';

    // Delete the withdrawal row
    sheet.deleteRow(i + 1);

    // Restore saving
    if (sourceSavingId) {
      // Partial withdrawal: find existing saving and add back amount
      var rows2 = sheet.getDataRange().getValues();
      for (var j = 1; j < rows2.length; j++) {
        if (String(rows2[j][0]) === sourceSavingId && rows2[j][5] === 'saving') {
          var newAmt = Number(rows2[j][3]) + withdrawnAmt;
          sheet.getRange(j + 1, 4).setValue(newAmt);
          sheet.getRange(j + 1, 5).setValue((newAmt / rate).toFixed(2));
          return jsonResponse({ ok: true, updatedSavingId: sourceSavingId, newAmt: newAmt });
        }
      }
    }
    // Full withdrawal (saving was deleted): recreate saving row
    var newId = String(Date.now() + 1);
    sheet.appendRow([newId, savedMonth, desc, withdrawnAmt, (withdrawnAmt / rate).toFixed(2), 'saving', '', '']);
    return jsonResponse({ ok: true, newSavingId: newId });
  }
  return jsonResponse({ error: 'Withdrawal entry not found' });
}

// ── RATE ──────────────────────────────────────────────────────────────────────

function getRate_() {
  const props = PropertiesService.getScriptProperties();
  return { rate: parseInt(props.getProperty('rate') || '90000') };
}

function getRate() {
  return jsonResponse(getRate_());
}

function setRate(body) {
  PropertiesService.getScriptProperties().setProperty('rate', String(body.rate));
  return jsonResponse({ ok: true });
}

// ── UTIL ──────────────────────────────────────────────────────────────────────

function jsonResponse(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
4
Save the script
~5 sec
→
Save using keyboard or toolbar

Press Ctrl+S (Windows) or Cmd+S (Mac), or click the floppy disk icon in the toolbar.

5
Run once to grant permissions
~1 min
→
Select a function to run

In the toolbar, make sure the function dropdown shows doGet or any function name (e.g. getAll).

→
Click the Run button

Click the ▶ Run button. Google will ask for permissions — click Review permissions, choose your account, then click Advanced → Go to Masraf (unsafe) → Allow.

This warning is normal — it just means the script was not published by Google. It is your own code running in your own account.
6
Create a new deployment
~1 min
→
Open the Deploy menu

In the Apps Script editor, click Deploy in the top-right area, then choose New deployment.

7
Select Web app type
~15 sec
→
Choose the deployment type

Click the gear ⚙ icon next to Select type and choose Web app.

8
Configure access settings
~15 sec
→
Execute as: Me

Under Execute as, select Me (your Google account).

→
Who has access: Anyone

Under Who has access, select Anyone. This allows the app to call the backend without requiring sign-in.

Note: Selecting Anyone makes the URL publicly accessible. Do not share the URL with people you do not trust, as it gives write access to your spreadsheet.
9
Deploy and copy the URL
~30 sec
→
Click Deploy

Click the Deploy button. If prompted again for permissions, follow the same steps as before and click Allow.

→
Copy the Web app URL

A dialog will appear with a Web app URL. It looks like:

https://script.google.com/macros/s/AKfycb.../exec

Click Copy to copy this URL — you will need it in the next step.

10
Enter the URL in Masraf
~30 sec
→
Open the Masraf app

Open the Masraf PWA in your browser.

→
Paste the URL

In the app settings or first-run prompt, paste your Web app URL and confirm. The app will connect to your Google Sheet and load your data.

Tip: Your URL is saved in the browser's local storage. You only need to enter it once per device.
